最长有效括号

Tim 2022-05-30 18:58:22
Categories： Tags： dp

一

dp[i][j] 表示从数组的 [0,i]下标范围内选取若干个正整数（可以是 0 个），是否存在一种选取方案使得被选取的正整数的和恰好等于 j。

class Solution {
    public boolean canPartition(int[] nums) {
        int n = nums.length;
        int sum = 0;
        int max = 0;
        for (int i : nums) {
            sum += i;
            max = Math.max(i, max);
        }
        if (sum % 2 != 0) {
            return false;
        }
        int half = sum / 2;
        if (max > half) {
            return false;
        }
        boolean[][] dp = new boolean[n][half + 1];
        dp[0][nums[0]] = true;
        for (int i = 1; i < n; i++) {
            int num = nums[i];
            for (int j = 1; j <= half; j++) {
                if (j >= num) {
                    // 选num
                    dp[i][j] = dp[i - 1][j] || dp[i - 1][j - num];
                } else {
                    // 不选num
                    dp[i][j] = dp[i - 1][j];
                }
            }
        }
        return dp[n - 1][half];
    }
}
